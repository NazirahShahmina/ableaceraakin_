<?php
session_start();
// Security gateway access validation check
if (!isset($_SESSION['admin_logged_in'])) {
    header("Location: login.php");
    exit;
}

$host = 'localhost';
$dbname = 'website_db';
$username = 'root';
$password = '';
$success_msg = "";
$error_msg = "";

try {
    $pdo = new PDO("mysql:host=$host;dbname=$dbname;charset=utf8mb4", $username, $password);
    $pdo->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);
} catch (PDOException $e) {
    die("Database connection error occurred.");
}

if ($_SERVER["REQUEST_METHOD"] == "POST") {
    $title = $_POST['title'];
    $category = $_POST['category'];
    $content = $_POST['content'];
    $author = $_SESSION['admin_user'] ?? 'Admin'; // Uses logged-in username
    $image_url = 'assets/img/default.jpg'; // Fallback if no image uploaded

    // MANAGE FILE UPLOAD ENGINE ROUTINE
    if (isset($_FILES['article_image']) && $_FILES['article_image']['error'] == 0) {
        $allowed_types = ['image/jpeg', 'image/png', 'image/gif', 'image/webp'];
        $file_type = $_FILES['article_image']['type'];
        
        if (in_array($file_type, $allowed_types)) {
            // Ensure uploads directory exists
            if (!is_dir('uploads')) {
                mkdir('uploads', 0777, true);
            }
            
            $filename = time() . '_' . basename($_FILES['article_image']['name']);
            $target_path = 'uploads/' . $filename;
            
            if (move_uploaded_file($_FILES['article_image']['tmp_name'], $target_path)) {
                $image_url = $target_path; // Update image path to point to our local file location
            }
        } else {
            $error_msg = "<div class='alert alert-danger'>Invalid file type. Only JPG, PNG, GIF, and WEBP allowed.</div>";
        }
    }

    if (empty($error_msg)) {
        // INSERT NEW RECORD EXECUTOR (Notice we don't lock to id = 1)
        $insert = $pdo->prepare("INSERT INTO articles (title, category, content, image_url, author) VALUES (:title, :category, :content, :image_url, :author)");
        $insert->execute([
            ':title' => $title,
            ':category' => $category,
            ':content' => $content,
            ':image_url' => $image_url
            ':author' => $author
        ]);

        $success_msg = "<div class='alert alert-success shadow-sm mb-4'>🎉 New article successfully published live!</div>";
    }
}

                <a href="logout.php" class="btn btn-sm btn-outline-danger ms-3">Sign Out</a>
            </div>
        </div>
    </nav>

    <div class="container" style="max-width: 800px;">
        <h2 class="mb-4 text-dark fw-bold">Publish a New Article</h2>
        
        <?php echo $success_msg; ?>
        <?php echo $error_msg; ?>

        <form action="admin-add.php" method="POST" enctype="multipart/form-data" class="bg-white p-4 rounded shadow-sm mb-5">
            
            <div class="mb-3">
                <label class="form-label fw-bold">Article Title</label>
                <input type="text" name="title" class="form-control" placeholder="Enter a professional headline" required>
            </div>

            <div class="mb-3">
                <label class="form-label fw-bold">Section Category</label>
                <select name="category" class="form-select" required>
                    <option value="Media Exposure">Media Exposure</option>
                    <option value="Latest News">Latest News</option>
                    <option value="Announcements">Announcements</option>
                </select>
            </div>

            <div class="mb-3">
                <label class="form-label fw-bold">Upload Featured Image</label>
                <input type="file" name="article_image" class="form-control" accept="image/*" required>
                <div class="form-text text-muted small">Choose a photo from your computer or phone storage.</div>
            </div>

            <div class="mb-3">
                <label class="form-label fw-bold">Article Body Content (Accepts raw HTML structure)</label>
                <textarea name="content" class="form-control" rows="12" placeholder="Write or paste your HTML formatted article here..." required></textarea>
            </div>

            <button type="submit" class="btn btn-primary w-100 fw-bold py-2">
                <i class="fas fa-paper-plane me-2"></i>Publish Live to Website
            </button>
        </form>
    </div>

</body>
